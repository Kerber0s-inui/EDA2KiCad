# 转换任务编排与产物交付 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立统一的转换任务编排层，支持用户指定输出目录、ZIP 输入、组合输入工程复用，以及只保留最终 `.kicad_pcb`、`.kicad_sch`、`.kicad_pro`、`report.json`。

**Architecture:** 在现有 strategy 之上增加 `jobs` 编排层，由它负责输入解析、任务目录创建、ZIP 解压、组合判断、子任务调度、结果汇总和中间目录清理。Web 和 CLI 统一改为调用该编排层，而不是各自直接拼装单次 `ConversionService` 调用。

**Tech Stack:** Python 3.11、FastAPI、Typer、pathlib、zipfile、pytest

from pathlib import Path

from eda2kicad.jobs.models import JobWorkspace


def create_job_workspace(output_dir: Path, input_label: str) -> JobWorkspace:
    stamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    root = output_dir / f"{stamp}_{input_label}"
    temp_dir = root / "_tmp"
    input_dir = temp_dir / "input"
    for path in (root, temp_dir, input_dir):
        path.mkdir(parents=True, exist_ok=True)
    return JobWorkspace(root=root, final_dir=root, temp_dir=temp_dir, input_dir=input_dir)

from pathlib import Path
import zipfile

from eda2kicad.jobs.models import PlannedInputs


def plan_job_inputs(input_path: Path, extract_root: Path) -> PlannedInputs:
    if input_path.suffix.lower() != ".zip":
        suffix = input_path.suffix.lower()
        return PlannedInputs(
            input_mode="single",
            label=input_path.stem,
            prjpcb=input_path if suffix == ".prjpcb" else None,
            pcbdoc=input_path if suffix == ".pcbdoc" else None,
            schdoc=input_path if suffix == ".schdoc" else None,
        )
    extract_root.mkdir(parents=True, exist_ok=True)
    with zipfile.ZipFile(input_path) as zf:
        zf.extractall(extract_root)
    files = list(extract_root.rglob("*"))
    prjpcb = next((p for p in files if p.suffix.lower() == ".prjpcb"), None)
    pcbdoc = next((p for p in files if p.suffix.lower() == ".pcbdoc"), None)
    schdoc = next((p for p in files if p.suffix.lower() == ".schdoc"), None)
    return PlannedInputs(input_mode="zip", label=input_path.stem, prjpcb=prjpcb, pcbdoc=pcbdoc, schdoc=schdoc)
```

# src/eda2kicad/jobs/planner.py
def choose_job_mode(*, prjpcb: Path | None, pcbdoc: Path | None, schdoc: Path | None) -> str:
    if prjpcb is not None:
        return "reuse-project"
    if pcbdoc is not None and schdoc is not None:
        return "shared-empty-project"
    if pcbdoc is not None or schdoc is not None:
        return "single-empty-project"
    raise ValueError("no supported input files found")

# src/eda2kicad/jobs/runner.py
from pathlib import Path


def collect_final_outputs(final_dir: Path) -> dict[str, Path]:
    for cache_lib in final_dir.glob("*-cache.lib"):
        cache_lib.unlink()
    outputs: dict[str, Path] = {}
    board = next(iter(sorted(final_dir.glob("*.kicad_pcb"))), None)
    schematic = next(iter(sorted(final_dir.glob("*.kicad_sch"))), None)
    project = next(iter(sorted(final_dir.glob("*.kicad_pro"))), None)
    if board is not None:
        outputs["board"] = board
    if schematic is not None:
        outputs["schematic"] = schematic
    if project is not None:
        outputs["project"] = project
    return outputs
```

def convert_native_bundle(
    *,
    pcb_input: Path | None,
    schematic_input: Path | None,
    output_root: Path | None = None,
) -> ConversionPayload:
    if pcb_input is None and schematic_input is None:
        raise ValueError("bundle import requires pcb_input or schematic_input")
    gui_output_root = resolve_strategy_work_root("kicad-gui-official", output_root)
    result = run_combined_gui_import(
        pcb_input=pcb_input,
        schematic_input=schematic_input,
        output_root=gui_output_root,
        kicad_exe=KICAD_GUI_PATH,
    )
    return {
        "project_name": str(result["project_name"]),
        "schematic_text": Path(result["schematic_path"]).read_text(encoding="utf-8") if result.get("schematic_path") else None,
        "schematic_extension": ".kicad_sch" if result.get("schematic_path") else None,
        "board_text": Path(result["board_path"]).read_text(encoding="utf-8") if result.get("board_path") else None,
        "board_extension": ".kicad_pcb" if result.get("board_path") else None,
        "auxiliary_text_artifacts": {
